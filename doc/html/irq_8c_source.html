42 /* Each of the IRQ ISRs point to this function, rather than
43  * the 'fault_handler' in 'isrs.c'. The IRQ Controllers need
44  * to be told when you are done servicing them, so you need
45  * to send them an "End of Interrupt" command (0x20). There
46  * are two 8259 chips: The first exists at 0x20, the second
47  * exists at 0xA0. If the second controller (an IRQ from 8 to
48  * 15) gets an interrupt, you need to acknowledge the
49  * interrupt at BOTH controllers, otherwise, you only send
50  * an EOI command to the first controller. If you don't send
51  * an EOI, you won't raise any more IRQs
52  */
53 void _irq_handler(regs_t * regs) {
54  uint8_t irq = regs->int_num - 32;
55 
56  // Get the handler
57  irq_handler handler = irq_handlers[irq];
58 
59  // Run the handler if got one
60  if (handler) {
61  handler(regs);
62  }
63 
64  // Send the end of interrupt command
65  pic_send_end_of_interrupt(irq);
66 }
